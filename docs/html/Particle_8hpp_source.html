166template <typename T, typename Fun>
167Particle<T, Fun>::Particle(const Fun objectiveFunction, const size_t &dimension,
168 std::mt19937 &rng,
169 std::vector<std::uniform_real_distribution<T>> &dis,
170 std::vector<T> lowerBounds,
171 std::vector<T> upperBounds)
172 : _dimension(dimension),
173 _position(dimension),
174 _velocity(dimension),
175 _bestPosition(dimension)
176{
177 double velocityScale = 0.1; // 10% of position range
178
179 for (size_t i = 0; i < _dimension; ++i)
180 {
181 _position[i] = dis[i](rng);
182 _velocity[i] = velocityScale * (dis[i](rng)-0.5);
183 }
184
185 _bestPosition = _position;
186 _value = objectiveFunction(_position.data());
187 _bestValue = _value;
188}
