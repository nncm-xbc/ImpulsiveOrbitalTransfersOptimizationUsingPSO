43bool Shader::loadFromSource(const std::string& vertexCode, const std::string& fragmentCode) {
44 const char* vShaderCode = vertexCode.c_str();
45 const char* fShaderCode = fragmentCode.c_str();
46
47 unsigned int vertex, fragment;
48
49 // Vertex Shader
50 vertex = glCreateShader(GL_VERTEX_SHADER);
51 glShaderSource(vertex, 1, &vShaderCode, NULL);
52 glCompileShader(vertex);
53 if (!checkCompileErrors(vertex, "VERTEX")) return false;
54
55 // Fragment Shader
56 fragment = glCreateShader(GL_FRAGMENT_SHADER);
57 glShaderSource(fragment, 1, &fShaderCode, NULL);
58 glCompileShader(fragment);
59 if (!checkCompileErrors(fragment, "FRAGMENT")) return false;
60
61 // Shader Program
62 ID = glCreateProgram();
63 glAttachShader(ID, vertex);
64 glAttachShader(ID, fragment);
65 glLinkProgram(ID);
66 if (!checkCompileErrors(ID, "PROGRAM")) return false;
67
68 glDeleteShader(vertex);
69 glDeleteShader(fragment);
70
71 return true;
72}
