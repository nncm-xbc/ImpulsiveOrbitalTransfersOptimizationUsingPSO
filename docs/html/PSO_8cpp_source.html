89template <typename T, typename Fun>
90void PSO<T, Fun>::updateWC(std::vector<T> GBPos_previous, size_t iter)
91{
92 // Linear decrease of inertia weight with iteration count
93 T currentIteration = static_cast<T>(iter);
94 T maxWeight = 0.9;
95 T minWeight = 0.4;
96 T inertiaWeight = maxWeight - ((maxWeight - minWeight) * currentIteration / _maxIterations);
97
98 swarm.setInertiaWeight(inertiaWeight);
99}
