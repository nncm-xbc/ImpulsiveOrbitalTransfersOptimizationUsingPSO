161bool loadPSOResultsFromFile(const std::string& filename, PSOOrbitTransferResult& result) {
162 std::ifstream file(filename);
163 if (!file.is_open()) {
164 std::cerr << "Failed to open PSO results file: " << filename << std::endl;
165 return false;
166 }
167
168 std::string line;
169 std::string section;
170
171 while (std::getline(file, line)) {
172 // Skip empty lines and comments
173 if (line.empty() || line[0] == '#')
174 continue;
175
176 // Check for section headers
177 if (line[0] == '[' && line.back() == ']') {
178 section = line.substr(1, line.size() - 2);
179 continue;
180 }
181
182 // Parse key-value pairs
183 std::istringstream iss(line);
184 std::string key;
185 std::string equals;
186 std::string value;
187
188 if (std::getline(iss, key, '=') && std::getline(iss, value)) {
189 // Trim whitespace
190 key.erase(0, key.find_first_not_of(" \t"));
191 key.erase(key.find_last_not_of(" \t") + 1);
192 value.erase(0, value.find_first_not_of(" \t"));
193 value.erase(value.find_last_not_of(" \t") + 1);
194
195 // Process based on section
196 if (section == "InitialOrbit") {
197 if (key == "radius" || key == "semi_major_axis")
198 result.initial_radius = std::stod(value);
199 else if (key == "inclination")
200 result.initial_inclination = std::stod(value);
201 else if (key == "raan")
202 result.initial_raan = std::stod(value);
203 else if (key == "eccentricity")
204 result.initial_eccentricity = std::stod(value);
205 else if (key == "arg_periapsis")
206 result.initial_arg_periapsis = std::stod(value);
207 }
208 else if (section == "TargetOrbit") {
209 if (key == "radius" || key == "semi_major_axis")
210 result.target_radius = std::stod(value);
211 else if (key == "inclination")
212 result.target_inclination = std::stod(value);
213 else if (key == "raan")
214 result.target_raan = std::stod(value);
215 else if (key == "eccentricity")
216 result.target_eccentricity = std::stod(value);
217 else if (key == "arg_periapsis")
218 result.target_arg_periapsis = std::stod(value);
219 }
220 else if (section == "OptimalTransfer") {
221 if (key == "initial_true_anomaly")
222 result.initial_true_anomaly = std::stod(value);
223 else if (key == "final_true_anomaly")
224 result.final_true_anomaly = std::stod(value);
225 else if (key == "transfer_time")
226 result.transfer_time = std::stod(value);
227 else if (key == "is_three_impulse")
228 result.is_three_impulse = (value == "true" || value == "1");
229 else if (key == "intermediate_radius")
230 result.intermediate_radius = std::stod(value);
231 }
232 else if (section == "DeltaV") {
233 if (key == "magnitude") {
234 std::istringstream vss(value);
235 std::string item;
236 result.delta_v_magnitudes.clear();
237 while (std::getline(vss, item, ',')) {
238 result.delta_v_magnitudes.push_back(std::stod(item));
239 }
240 }
241 if (key == "plane_change") {
242 std::istringstream vss(value);
