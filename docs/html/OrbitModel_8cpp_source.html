109void OrbitModel::generateOrbitPoints(int resolution) {
110 orbit_points_.clear();
111 std::vector<glm::vec3> physics_points;
112
113 for (int i = 0; i < resolution; ++i) {
114 double true_anomaly = 2.0 * M_PI * i / (resolution - 1);
115
116 double r;
117 if (eccentricity_ < 1e-6) { // Circular orbit
118 r = radius_;
119 } else { // Elliptical orbit
120 double p = radius_ * (1.0 - eccentricity_ * eccentricity_);
121 r = p / (1.0 + eccentricity_ * cos(true_anomaly));
122 }
123
124 // Position in orbital plane
125 glm::vec3 pos_orbital(
126 r * cos(true_anomaly),
127 r * sin(true_anomaly),
128 0.0
129 );
130
131 // Transform to inertial physics coordinates
132 glm::mat3 R = CoordinateSystem::orbitalToInertialMatrix(
133 raan_, inclination_, arg_periapsis_
134 );
135
136 glm::vec3 pos_physics = R * pos_orbital;
137 physics_points.push_back(pos_physics);
138
139 }
140
141 orbit_points_ = CoordinateSystem::trajectoryToVisualization(physics_points);
142
143 if (physics_points.size() >= 3) {
144 glm::vec3 expected_normal(
145 sin(raan_) * sin(inclination_),
146 -cos(raan_) * sin(inclination_),
147 cos(inclination_)
148 );
149 }
150}
