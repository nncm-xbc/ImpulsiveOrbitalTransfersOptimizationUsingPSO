32template <typename T, typename Fun>
33void Swarm<T, Fun>::init()
34{
35 std::random_device rd;
36 _rng = std::mt19937(rd());
37 _dis = std::vector<std::uniform_real_distribution<T>>(_dimension);
38 _gBestVal = std::numeric_limits<T>::max();
39 _gBestPos.resize(_dimension, std::numeric_limits<double>::quiet_NaN());
40
41 double margin = 0.05;
42
43 for (size_t i = 0; i < _dimension; ++i)
44 {
45 double range = _upperBounds[i] - _lowerBounds[i];
46 double adjustedLower = _lowerBounds[i] + (margin * range);
47 double adjustedUpper = _upperBounds[i] - (margin * range);
48 _dis[i] = (std::uniform_real_distribution<T>(adjustedLower, adjustedUpper));
49 }
50
51 for(size_t i = 0; i < _numParticles; ++i) {
52 particles.emplace_back(Particle<T, Fun>(_objectiveFunction, _dimension, _rng, _dis, _lowerBounds, _upperBounds));
53 }

55}

58void Swarm<T, Fun>::updatePosition(Particle<T, Fun> &particle)
59{
60 std::vector<T> newPosition = particle.getPosition();

62
63 for (size_t i = 0; i < _dimension; ++i)
64 {
65
66 newPosition[i] += velocity[i];
67
68 // boundary checks
69 if (newPosition[i] < _lowerBounds[i])
70 {
71 newPosition[i] = _lowerBounds[i];
72 std::vector<T> newVelocity = particle.getVelocity();

78 newPosition[i] = _upperBounds[i];
79
80 std::vector<T> newVelocity = particle.getVelocity();

83 }
84 else if (std::isnan(newPosition[i]) || std::isinf(newPosition[i]))
85 {
86 // Reset to a random value within bounds
87 newPosition[i] = _lowerBounds[i] + (_upperBounds[i] - _lowerBounds[i]) * ((double)rand() / RAND_MAX);
88 std::vector<T> newVelocity = particle.getVelocity();
89 newVelocity[i] = 0.0;

91 }

96template <typename T, typename Fun>
97void Swarm<T, Fun>::updateVelocity(Particle<T, Fun> &particle)
98{
99
100 std::vector<T> newVelocity(particle.getVelocity());
101 const std::vector<T> &position = particle.getPosition();
102 const std::vector<T> &pBest = particle.getBestPosition();
103
104 for (size_t i = 0; i < _dimension; ++i)
105 {
106 T r1 = _dis[i](_rng);
107 T r2 = _dis[i](_rng);
108
109 if (position[i] + newVelocity[i] < _lowerBounds[i] || position[i] + newVelocity[i] > _upperBounds[i])
110 {
111 newVelocity[i] = 0.0;
112 } else {
113
114 newVelocity[i] = _inertiaWeight * newVelocity[i] +
115 _cognitiveWeight * r1 * (pBest[i] - position[i]) +
116 _socialWeight * r2 * (_gBestPos[i] - position[i]);
117 }
118 }
119 particle.setVelocity(newVelocity);
120}
