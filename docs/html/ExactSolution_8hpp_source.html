82 Solution compute() const {

88 T v2 = std::sqrt(_MU/_R2);

94 sol.deltaV2 = v2 - std::sqrt(_MU*(2.0/_R2 - 1.0/a));
95 sol.deltaV = std::abs(sol.deltaV1) + std::abs(sol.deltaV2);
