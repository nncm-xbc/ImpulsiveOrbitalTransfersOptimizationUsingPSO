144template <typename T, typename Fun>
145double OrbitTransferObjective<T, Fun>::computePeriapsis(const std::vector<T>& x)
146{
147 double departureTrueAnomaly = x[0];
148 double firstImpulseMagnitude = x[2];
149 double firstImpulseDirection = x[4];
150
151 // Radius at departure point
152 double r_departure = _R1 * (1 - _e1*_e1) / (1 + _e1 * std::cos(departureTrueAnomaly));
153
154 auto v_init = Physics::OrbitMechanics::calculateVelocity(_R1, _e1, departureTrueAnomaly);
155
156 // Post-impulse velocity components
157 double v1_r = v_init.first + firstImpulseMagnitude * std::sin(firstImpulseDirection);
158 double v1_theta = v_init.second + firstImpulseMagnitude * std::cos(firstImpulseDirection);
159
160 // Transfer orbit parameters
161 double vt = std::sqrt(v1_r*v1_r + v1_theta*v1_theta);
162 double h = r_departure * v1_theta;
163 double e = std::sqrt(1 + (h*h/(_MU*r_departure))*(vt*vt/(2*_MU) - 1/r_departure));
164 double a = h*h/(_MU*(1-e*e));
165
166 return a * (1-e);
167}

194template <typename T, typename Fun>
195double OrbitTransferObjective<T, Fun>::checkConstraints(const std::vector<double>& x)
196{
197 double totalViolation = 0.0;
198 double relaxationFactor = getRelaxationFactor();
199
200 // Check parameter bounds
201 if (x[0] < 0) totalViolation += std::abs(x[0]);
202 if (x[0] > 2*M_PI) totalViolation += (x[0] - 2*M_PI);
203 if (x[1] < 0) totalViolation += std::abs(x[1]);
204 if (x[1] > 2*M_PI) totalViolation += (x[1] - 2*M_PI);
205 if (x[2] < 0) totalViolation += std::abs(x[2]);
206 if (x[5] <= 0) totalViolation += (0.1 - x[5]);
207
208 // Non-coplanar case constraints
209 if (_i1 != 0 || _i2 != 0) {
210 Physics::Vector3 r_init = Physics::OrbitMechanics::calculatePosition3D(_R1, _e1, _i1, _raan1, _omega1, x[0]);
211 Physics::Vector3 r_final = Physics::OrbitMechanics::calculatePosition3D(_R2, _e2, _i2, _raan2, _omega2, x[1]);
212
213 double r_init_mag = r_init.magnitude();
214 double r_final_mag = r_final.magnitude();
215 double semi_major = (r_init_mag + r_final_mag) / 2.0;
216
217 // Transfer time constraints
218 double min_time = M_PI * sqrt(pow(semi_major, 3) / _MU);
219 if (x[5] < min_time / relaxationFactor)
220 totalViolation += (min_time / relaxationFactor - x[5]);
221
222 double max_time = 5.0 * min_time;
223 if (x[5] > max_time / relaxationFactor)
224 totalViolation += (x[5] - max_time / relaxationFactor);
225
226 // Impulse magnitude constraints
227 double escape_velocity = sqrt(2.0 * _MU / r_init_mag);
228 if (x[2] > escape_velocity)
229 totalViolation += (x[2] - escape_velocity);
230 if (x[3] > escape_velocity)
231 totalViolation += (x[3] - escape_velocity);
232
233 // Impulse ratio constraint
234 double impulse_ratio = std::max(x[2], x[3]) / (std::min(x[2], x[3]) + 1e-10);
235 if (impulse_ratio > 20.0 / relaxationFactor)
236 totalViolation += (impulse_ratio - 20.0 / relaxationFactor);
237
238 // Max impulse magnitude constraints
239 double v_circ1 = sqrt(_MU / _R1);
240 double v_circ2 = sqrt(_MU / _R2);
241 double v_max = 2.0 * std::max(v_circ1, v_circ2);
242
243 if (x[2] > v_max * relaxationFactor)
244 totalViolation += (x[2] - v_max * relaxationFactor);
245 if (x[3] > v_max * relaxationFactor)
246 totalViolation += (x[3] - v_max * relaxationFactor);
247
248 // Min impulse magnitude constraints
249 double min_impulse = 1e-4 * std::min(v_circ1, v_circ2);
250 if (x[2] < min_impulse / relaxationFactor)
251 totalViolation += (min_impulse / relaxationFactor - x[2]);
252 if (x[3] < min_impulse / relaxationFactor)
253 totalViolation += (min_impulse / relaxationFactor - x[3]);
254
255 // Plane change constraint
256 if (_i1 != _i2) {
257 double min_plane_change_dv = 0.8 * v_circ1 * sin(fabs(_i1 - _i2) / 2.0);
258 if (x[2] + x[3] < min_plane_change_dv / relaxationFactor)
259 totalViolation += (min_plane_change_dv / relaxationFactor - (x[2] + x[3]));
260 }
261
262 // Intersection constraint
263 if (!doesIntersect(x)) {
264 totalViolation += 200.0;
265 }
266
267 }
268 else {
269 // Coplanar case
270 double transferPeriapsis = computePeriapsis(x);
271 double transferApoapsis = computeApoapsis(x);
272
273 if (transferPeriapsis > std::min(_R1, _R2))
274 totalViolation += (transferPeriapsis - std::min(_R1, _R2));
275 if (transferApoapsis < std::max(_R1, _R2))
276 totalViolation += (std::max(_R1, _R2) - transferApoapsis);
277 if (transferApoapsis > _Rmax)
278 totalViolation += (transferApoapsis - _Rmax);
279 }
280
281 return totalViolation;
282}

285template<typename T, typename Fun>
286bool OrbitTransferObjective<T, Fun>::doesIntersect(const std::vector<double>& x) {
287 double departureTrueAnomaly = x[0];
288 double firstImpulseMagnitude = x[2];
289 double firstImpulseDirection = x[4];
290
291 Physics::Vector3 r_init = Physics::OrbitMechanics::calculatePosition3D(_R1, _e1, _i1, _raan1, _omega1, departureTrueAnomaly);
292 Physics::Vector3 v_init = Physics::OrbitMechanics::calculateVelocity3D(_R1, _e1, _i1, _raan1, _omega1, departureTrueAnomaly);
293
294 Physics::Vector3 impulse = Physics::OrbitMechanics::calculateImpulseVector(
295 Physics::Vector3(r_init.x, r_init.y, r_init.z),
296 Physics::Vector3(v_init.x, v_init.y, v_init.z),
297 firstImpulseMagnitude,
298 firstImpulseDirection
299 );
300 Physics::Vector3 v_transfer = v_init;
301 v_transfer.x += impulse.x;
302 v_transfer.y += impulse.y;
303 v_transfer.z += impulse.z;
304
305 // Orbital elements
306 double mu = _MU;
307 double r_mag = sqrt(r_init.x*r_init.x + r_init.y*r_init.y + r_init.z*r_init.z);
308 double v_mag = sqrt(v_transfer.x*v_transfer.x + v_transfer.y*v_transfer.y + v_transfer.z*v_transfer.z);
309
310 // Specific angular momentum
311 Physics::Vector3 h;
312 h.x = r_init.y * v_transfer.z - r_init.z * v_transfer.y;
313 h.y = r_init.z * v_transfer.x - r_init.x * v_transfer.z;
314 h.z = r_init.x * v_transfer.y - r_init.y * v_transfer.x;
315 double h_mag = sqrt(h.x*h.x + h.y*h.y + h.z*h.z);
316
317 // Eccentricity vector and magnitude
318 double r_dot_v = r_init.x*v_transfer.x + r_init.y*v_transfer.y + r_init.z*v_transfer.z;
319 Physics::Vector3 e_vec;
320 e_vec.x = ((v_mag*v_mag - mu/r_mag) * r_init.x - r_dot_v * v_transfer.x) / mu;
321 e_vec.y = ((v_mag*v_mag - mu/r_mag) * r_init.y - r_dot_v * v_transfer.y) / mu;
322 e_vec.z = ((v_mag*v_mag - mu/r_mag) * r_init.z - r_dot_v * v_transfer.z) / mu;
323 double ecc = sqrt(e_vec.x*e_vec.x + e_vec.y*e_vec.y + e_vec.z*e_vec.z);
324
325 // Semi-major axis
326 double sma = h_mag*h_mag / (mu * (1.0 - ecc*ecc));
327
328 double periapsis = sma * (1.0 - ecc);
329 double apoapsis = sma * (1.0 + ecc);
330
331 // Check if target orbit is between periapsis and apoapsis
332 return (_R2 >= periapsis - 1e-6 && _R2 <= apoapsis + 1e-6);
333}
