15Vector3 OrbitMechanics::calculateVelocity3D(double a, double e, double i, double raan,
16 double omega, double true_anomaly)
17{
18 // Trigonometric values
19 const double cos_nu = cos(true_anomaly);
20 const double sin_nu = sin(true_anomaly);
21 const double cos_omega = cos(omega);
22 const double sin_omega = sin(omega);
23 const double cos_i = cos(i);
24 const double sin_i = sin(i);
25 const double cos_raan = cos(raan);
26 const double sin_raan = sin(raan);
27
28 const double p = a * (1 - e * e);
29 const double r = p / (1 + e * cos_nu);
30 const double sqrt_mu_p = sqrt(constant::MU / p);
31
32 // Velocity components in perifocal frame (radial and transverse)
33 const double v_r = sqrt_mu_p * e * sin_nu;
34 const double v_theta = sqrt_mu_p * (1 + e * cos_nu);
35
36 // Combined rotation matrix application (perifocal to inertial frame)
37 const double v_P = v_r * cos_nu - v_theta * sin_nu;
38 const double v_Q = v_r * sin_nu + v_theta * cos_nu;
39
40 // Rotations
41 double v_x = (v_P * cos_omega - v_Q * sin_omega) * cos_raan -
42 ((v_P * sin_omega + v_Q * cos_omega) * cos_i) * sin_raan;
43 double v_y = (v_P * cos_omega - v_Q * sin_omega) * sin_raan +
44 ((v_P * sin_omega + v_Q * cos_omega) * cos_i) * cos_raan;
45 double v_z = (v_P * sin_omega + v_Q * cos_omega) * sin_i;
46
47 return Vector3(v_x, v_y, v_z);
48}

65
66 // Position in orbital plane (perifocal coordinates)

68 const double y_peri = r * sin_nu;
69
70 // Combined rotation matrix application (perifocal to inertial frame)

116Vector3 OrbitMechanics::calculateImpulseVector(const Vector3& pos, const Vector3& vel,
117 double magnitude, double planeChangeAngle)
118{
119 // Unit vectors for orbital frame
120 // Tangential = velocity direction
121 double v_mag = sqrt(vel.x*vel.x + vel.y*vel.y + vel.z*vel.z);
122 Vector3 t_hat = {vel.x/v_mag, vel.y/v_mag, vel.z/v_mag};
123
124 // Radial = position direction
125 double r_mag = sqrt(pos.x*pos.x + pos.y*pos.y + pos.z*pos.z);
126 Vector3 r_hat = {pos.x/r_mag, pos.y/r_mag, pos.z/r_mag};
127
128 Vector3 n_hat = {
129 r_hat.y * t_hat.z - r_hat.z * t_hat.y,
130 r_hat.z * t_hat.x - r_hat.x * t_hat.z,
131 r_hat.x * t_hat.y - r_hat.y * t_hat.x
132 };
133
134 // Normal vector
135 double n_mag = sqrt(n_hat.x*n_hat.x + n_hat.y*n_hat.y + n_hat.z*n_hat.z);
136 n_hat.x /= n_mag;
137 n_hat.y /= n_mag;
138 n_hat.z /= n_mag;
139
140 // Impulse components
141 double tangential_component = magnitude * cos(planeChangeAngle);
142 double normal_component = magnitude * sin(planeChangeAngle);
143
144 // Impulse vector = tangential component + normal component
145 Vector3 impulse = {
146 t_hat.x * tangential_component + n_hat.x * normal_component,
147 t_hat.y * tangential_component + n_hat.y * normal_component,
148 t_hat.z * tangential_component + n_hat.z * normal_component
149 };
150
151 return impulse;
152}
