10std::pair<glm::dvec3, glm::dvec3> LambertSolver::solveLambert(
11 const glm::dvec3& r1_glm,
12 const glm::dvec3& r2_glm,
13 double tof,
14 double mu,
15 bool isLongWay)
16{
17 Physics::Vector3 r1(r1_glm);
18 Physics::Vector3 r2(r2_glm);
19
20 // Position magnitudes and unit vectors
21 double r1_mag = r1.magnitude();
22 double r2_mag = r2.magnitude();
23 Physics::Vector3 r1_hat( r1.x / r1_mag, r1.y / r1_mag, r1.z / r1_mag);
24 Physics::Vector3 r2_hat( r2.x / r2_mag, r2.y / r2_mag, r2.z / r2_mag);
25
26 // Transfer angle
27 double cos_theta = (r1_hat.x * r2_hat.x + r1_hat.y * r2_hat.y + r1_hat.z * r2_hat.z);
28 cos_theta = std::min(1.0, std::max(-1.0, cos_theta));
29 double theta_tilde = std::acos(cos_theta);
30
31 // Transfer plane normal vector
32 Physics::Vector3 h_T(r1_hat.y * r2_hat.z - r1_hat.z * r2_hat.y,
33 r1_hat.z * r2_hat.x - r1_hat.x * r2_hat.z,
34 r1_hat.x * r2_hat.y - r1_hat.y * r2_hat.x);
35 double h_T_mag = h_T.magnitude();
36
37 if (h_T_mag < 1e-10) {
38 // For coplanar orbits, use z-axis as normal vector
39 h_T = Physics::Vector3(0, 0, 1);
40 } else {
41 h_T = Physics::Vector3(h_T.x / h_T_mag, h_T.y / h_T_mag, h_T.z / h_T_mag);
42 }
43
44 double theta = isLongWay ? 2 * M_PI - theta_tilde : theta_tilde;
45
46 // Geometric parameters
47 double chord = std::sqrt(r1_mag*r1_mag + r2_mag*r2_mag - 2*r1_mag*r2_mag*std::cos(theta));
48 double s = (r1_mag + r2_mag + chord) / 2.0;
49 double a_min = s / 2.0; // Minimum energy ellipse semi-major axis
50
51 // Minimum time of flight
52 double alpha_min = 2.0 * std::asin(std::sqrt((s - chord) / s));
53 if (theta > M_PI) {
54 alpha_min = 2.0 * M_PI - alpha_min;
55 }
56 double t_min = std::sqrt(a_min*a_min*a_min / mu) * (alpha_min - std::sin(alpha_min) - M_PI + std::sin(M_PI));
57
58 // Check if transfer is physically possible
59 if (tof < t_min) {
60 std::cout << "Error: Time of flight is less than minimum possible." << std::endl;
61 return std::make_pair(glm::dvec3(), glm::dvec3());
62 }
63
64 // Semi-major axis using bisection method
65 double a_lower = a_min;
66 double a_upper = a_min * 100.0;
67 double a = a_min;
68 double tof_current = t_min;
69
70 const int max_iter = 100;
71 const double tolerance = 1e-8;
72
73 for (int i = 0; i < max_iter && std::abs(tof_current - tof) > tolerance; ++i) {
74 if (tof_current < tof) {
75 a_lower = a;
76 } else {
77 a_upper = a;
78 }
79 a = (a_lower + a_upper) / 2.0;
80
81 // Time of flight for current semi-major axis
82 double alpha = 2.0 * std::asin(std::sqrt(s / (2.0 * a)));
83 double beta = 2.0 * std::asin(std::sqrt((s - chord) / (2.0 * a)));
84
85 if (theta > M_PI) {
86 alpha = 2.0 * M_PI - alpha;
87 }
88
89 tof_current = std::sqrt(a*a*a / mu) * (alpha - std::sin(alpha) - (beta - std::sin(beta)));
90 }
91
92 // Lagrange coefficients
93 double sin_theta = std::sin(theta);
94 double f, g, f_dot, g_dot;
95
96 if (std::abs(sin_theta) < 1e-10) {
97 // Special case handling for nearly 0° or 180° transfers
98 double delta = 1e-10;
99
100 if (std::abs(theta) < delta || std::abs(theta - 2*M_PI) < delta) {
101 // For θ ≈ 0° (direct transfer)
102 double h = r2_mag - r1_mag; // Altitude difference
103 f = 1.0 - h/a;
104 g = tof - sqrt(a*a*a/mu) * (theta - sin_theta);
105 g_dot = 1.0;
106 f_dot = -sqrt(mu/(a*a*a)) * h;
107 } else if (std::abs(theta - M_PI) < delta) {
108 // For θ ≈ 180° (transfer through central body)
109 f = -1.0;
110 g = sqrt(a*a*a/mu) * (M_PI - 2.0);
111 g_dot = -1.0;
112 f_dot = 0.0;
113 } else {
114 std::cerr << "Error in transfer angle calculation" << std::endl;
115 return std::make_pair(glm::dvec3(), glm::dvec3());
116 }
117 } else {
118 // Normal case
119 f = 1.0 - r2_mag / a * (1.0 - std::cos(theta));
120 g = r1_mag * r2_mag * sin_theta / std::sqrt(mu * a);
121 g_dot = 1.0 - r1_mag / a * (1.0 - std::cos(theta));
122 f_dot = -std::sqrt(mu / a) * sin_theta / (r1_mag * r2_mag) * r2_mag;
123 }
124
125 // Velocity vectors
126 Physics::Vector3 v1((r2.x - f * r1.x) / g, (r2.y - f * r1.y) / g, (r2.z - f * r1.z) / g);
127 Physics::Vector3 v2(f_dot * r1.x + g_dot * v1.x, f_dot * r1.y + g_dot * v1.y, f_dot * r1.z + g_dot * v1.z);
128
129 // Convert back to glm::dvec3
130 return std::make_pair(glm::dvec3(v1), glm::dvec3(v2));
131}

133std::optional<LambertSolver::Solution> LambertSolver::solve(
134 const glm::dvec3& r1,
135 const glm::dvec3& r2,
136 double tof,
137 const Config& config) const {
138
139 auto result = solveLambert(r1, r2, tof, mu_, config.long_way);
140
141 // Sanity check
142 if (glm::length(result.first) < 1e-10 && glm::length(result.second) < 1e-10) {
143 return std::nullopt;
144 }
145
146 Solution sol;
147 sol.v1 = result.first;
148 sol.v2 = result.second;
149 sol.is_valid = true;
150
151 // Semi-major axis and eccentricity
152 double r1_mag = glm::length(r1);
153 double v1_mag = glm::length(sol.v1);
154 double specific_energy = v1_mag * v1_mag / 2.0 - mu_ / r1_mag;
155 sol.semi_major_axis = -mu_ / (2.0 * specific_energy);
156
157 glm::dvec3 h = glm::cross(r1, sol.v1);
158 double h_mag = glm::length(h);
159 sol.eccentricity = std::sqrt(1.0 + 2.0 * specific_energy * h_mag * h_mag / (mu_ * mu_));
160
161 return sol;
162}

164double LambertSolver::calculateMinimumTOF(const glm::dvec3& r1,
165 const glm::dvec3& r2,
166 bool long_way) const {
167 double r1_mag = glm::length(r1);
168 double r2_mag = glm::length(r2);
169
170 glm::dvec3 r1_hat = r1 / r1_mag;
171 glm::dvec3 r2_hat = r2 / r2_mag;
172
173 double cos_theta = glm::dot(r1_hat, r2_hat);
174 double theta = std::acos(glm::clamp(cos_theta, -1.0, 1.0));
175
176 if (long_way) {
177 theta = 2 * M_PI - theta;
178 }
179
180 double chord = std::sqrt(r1_mag * r1_mag + r2_mag * r2_mag -
181 2 * r1_mag * r2_mag * std::cos(theta));
182 double s = (r1_mag + r2_mag + chord) / 2.0;
183 double a_min = s / 2.0;
184
185 double alpha = 2.0 * std::asin(std::sqrt(s / (2.0 * a_min)));
186 if (theta > M_PI) {
187 alpha = 2 * M_PI - alpha;
188 }
189
190 return std::sqrt(a_min * a_min * a_min / mu_) * (alpha - std::sin(alpha));
191}

193bool LambertSolver::validateSolution(const Solution& solution,
194 const glm::dvec3& r1,
195 const glm::dvec3& r2,
196 double expected_tof) const {
197 // Propagate from r1 with v1 and check if we reach r2
198
199 double r1_mag = glm::length(r1);
200 double r2_mag = glm::length(r2);
201
202 // Check energy consistency
203 double v1_mag = glm::length(solution.v1);
204 double v2_mag = glm::length(solution.v2);
205
206 double energy1 = v1_mag * v1_mag / 2.0 - mu_ / r1_mag;
207 double energy2 = v2_mag * v2_mag / 2.0 - mu_ / r2_mag;
208
209 if (std::abs(energy1 - energy2) > 1e-6) {
210 return false; // Energy not conserved
211 }
212
213 // Check angular momentum consistency
214 glm::dvec3 h1 = glm::cross(r1, solution.v1);
215 glm::dvec3 h2 = glm::cross(r2, solution.v2);
216
217 if (glm::length(h1 - h2) > 1e-6) {
218 return false; // Angular momentum not conserved
219 }
220
221 return true;
222}

224// Stumpff functions for universal variable formulation
225double LambertSolver::stumpffC(double z) const {
226 if (std::abs(z) < 1e-10) {
227 return 0.5;
228 } else if (z > 0) {
229 double sqrt_z = std::sqrt(z);
230 return (1.0 - std::cos(sqrt_z)) / z;
231 } else {
232 double sqrt_neg_z = std::sqrt(-z);
233 return (std::cosh(sqrt_neg_z) - 1.0) / (-z);
234 }
235}

237double LambertSolver::stumpffS(double z) const {
238 if (std::abs(z) < 1e-10) {
239 return 1.0 / 6.0;
240 } else if (z > 0) {
241 double sqrt_z = std::sqrt(z);
242 return (sqrt_z - std::sin(sqrt_z)) / (z * sqrt_z);
243 } else {
244 double sqrt_neg_z = std::sqrt(-z);
245 return (std::sinh(sqrt_neg_z) - sqrt_neg_z) / (-z * sqrt_neg_z);
246 }
