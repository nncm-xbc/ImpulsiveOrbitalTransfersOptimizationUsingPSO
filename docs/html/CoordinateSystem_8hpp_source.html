62inline glm::vec3 physicsToVisualization(const glm::vec3& physics_pos) {

133inline glm::mat3 orbitalToInertialMatrix(float raan, float inclination, float arg_periapsis) {
134 // Standard orbital mechanics rotation sequence
135 glm::mat3 R_raan = glm::mat3(
136 cos(raan), -sin(raan), 0.0f,
137 sin(raan), cos(raan), 0.0f,
138 0.0f, 0.0f, 1.0f
139 );
140
141 glm::mat3 R_inc = glm::mat3(
142 1.0f, 0.0f, 0.0f,
143 0.0f, cos(inclination), -sin(inclination),
144 0.0f, sin(inclination), cos(inclination)
145 );
146
147 glm::mat3 R_omega = glm::mat3(
148 cos(arg_periapsis), -sin(arg_periapsis), 0.0f,
149 sin(arg_periapsis), cos(arg_periapsis), 0.0f,
150 0.0f, 0.0f, 1.0f
151 );
152
153 // Combined transformation
154 return R_raan * R_inc * R_omega;
155}

166inline bool verifyOrbitalPlane(const std::vector<glm::vec3>& points,
167 const glm::vec3& expected_normal,
168 float tolerance = 1e-3f) {
169 if (points.size() < 3) return true;
170
171 glm::vec3 v1 = points[1] - points[0];
172 glm::vec3 v2 = points[2] - points[0];
173 glm::vec3 actual_normal = glm::normalize(glm::cross(v1, v2));
174
175 // Check if normals are aligned (or opposite)
176 float dot = glm::abs(glm::dot(actual_normal, expected_normal));
177 return dot > (1.0f - tolerance);
178}
